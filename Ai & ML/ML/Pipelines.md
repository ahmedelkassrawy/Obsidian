# Importing necessary libraries
import numpy as np
import pandas as pd

# Creating a dictionary with sample data
d1 = {
    'Sold_out': [1, 0, 0, 1, 0, 0, 0, 1, 0, 1],
    'Social_media_followers': [0, float('nan'), 2000000.0, 1310000.0, 1700000.0, float('nan'), 4100000.0, 1600000.0, 2200000.0, 1000000.0]
}

# Converting the dictionary to a pandas DataFrame
df1 = pd.DataFrame(data=d1)

# Selecting features (X) and target variable (y)
X1 = df1[["Social_media_followers"]]
y1 = df1[["Sold_out"]]

# Splitting the data into training and testing sets
from sklearn.model_selection import train_test_split
X1_train, X1_test, y1_train, y1_test = train_test_split(X1,
                                                        y1,
                                                        test_size=0.3,
                                                        random_state=19)

# Creating an imputer to handle missing values
from sklearn.impute import SimpleImputer
imputer = SimpleImputer(strategy="mean")

# Creating a logistic regression model
from sklearn.linear_model import LogisticRegression
lr = LogisticRegression()

# Creating a pipeline that combines imputer and logistic regression
from sklearn.pipeline import make_pipeline
pipe1 = make_pipeline(imputer, lr)

# Fitting the pipeline on the training data
pipe1.fit(X1_train, y1_train)

# Evaluating the model on training data
pipe1.score(X1_train, y1_train)

# Evaluating the model on test data
pipe1.score(X1_test, y1_test)

# Checking the imputation statistics
pipe1.named_steps.simpleimputer.statistics_

# Creating a more advanced pipeline with additional features
d2 = {'Genre':['Rock', 'Metal', 'Bluegrass', 'Rock', np.nan, 'Rock', 'Rock', np.nan, 'Bluegrass', 'Rock'],
    'Social_media_followers':[1000000, np.nan, 2000000, 1310000, 1700000, np.nan, 4100000, 1600000, 2200000, 1000000],
    'Sold_out':[1, 0, 0, 1, 0, 0, 0, 1, 0, 1]}

df = pd.DataFrame(data=d2)

# Selecting features (X) and target variable (y)
X = df.iloc[:,0:2]
y = df.iloc[:,2]

# Splitting the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=70)

# Defining numerical and categorical columns
nums_cols = ["Social_media_followers"]
cat_cols = ["Genre"]

# Importing necessary components for the pipeline
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler, OneHotEncoder

# Creating a pipeline for numerical features
num_pipeline = Pipeline(
    steps=[
        ("impute",SimpleImputer(strategy="mean")),
        ("scale", StandardScaler())
    ]
)

# Creating a pipeline for categorical features
cat_pipeline = Pipeline(
    steps=[
        ("impute", SimpleImputer(strategy="most_frequent")),
        ("one-hot-encoder", OneHotEncoder(handle_unknown ="ignore",sparse_output=False))
    ]
)

# Combining numerical and categorical pipelines
from sklearn.compose import ColumnTransformer
col_transformer = ColumnTransformer(
    transformers=[
        ("num", num_pipeline, nums_cols),
        ("cat", cat_pipeline, cat_cols)
    ],
    remainder="drop",
    n_jobs = -1
)

# Creating a decision tree classifier
from sklearn.tree import DecisionTreeClassifier
dtc = DecisionTreeClassifier()

# Creating the final pipeline
pipefinal = make_pipeline(col_transformer, dtc)

# Fitting the final pipeline on the training data
pipefinal.fit(X_train, y_train)

# Evaluating the model on test data
pipefinal.score(X_test, y_test)

# Saving the model to a file
import joblib
joblib.dump(pipefinal, "pipefinal.joblib")

# Loading the saved model
pipefinal2 = joblib.load("pipefinal.joblib")
