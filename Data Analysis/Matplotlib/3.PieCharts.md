```python
from matplotlib import pyplot as plt
plt.style.use("fivethirtyeight")
  
slices = [120,80,30,20]
labels = ["sixty","forty","Extra1","extra2"]
colors = ["#008fd5","#fc4f30","#e5ae37","#6d904f"]
  
plt.pie(slices,labels = labels, wedgeprops={"edgecolor":"black"}, colors = colors)
#first include the list of things to pie chart
#labels
#wedge props as the some props we here used the edge color only it take it as a dict
#colors should also be passed as a list


plt.title("My Awesome Pie Chart")
plt.tight_layout()
plt.show()
  
# Colors:
# Blue = #008fd5
# Red = #fc4f30
# Yellow = #e5ae37
# Green = #6d904f
```

```python
from matplotlib import pyplot as plt
  
plt.style.use("fivethirtyeight")

# Language Popularity
slices = [59219, 55466, 47544, 36443, 35917]
labels = ['JavaScript', 'HTML/CSS', 'SQL', 'Python', 'Java']
explode = [0,0,0,0.1,0] #explode is how far the slice in the pie chart will standout comparing to the others in same pie chart

plt.pie(slices,labels = labels, wedgeprops={"edgecolor":"black"}, explode = explode,shadow = True, startangle = 90, autopct="%1.1f%%") 
#shadow gives you the 3D effect
#start angle = 90 
#autopct is to have the percentage above each slice 

plt.title("My Awesome Pie Chart")
plt.tight_layout()
plt.show()
  
# Colors:
# Blue = #008fd5
# Red = #fc4f30
# Yellow = #e5ae37
# Green = #6d904f
```
